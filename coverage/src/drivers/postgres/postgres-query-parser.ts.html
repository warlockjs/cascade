    // correctly even when WHERE clauses appear before JOINs in the operations list.
    const JOIN_TYPES = new Set([

      "joinRaw",

    this.hasJoins = false;

    for (const operation of this.operations) {
      if (JOIN_TYPES.has(operation.type)) {

  }
